 # @date 03.04.2013

scai_project ( 

    CLASSES
    
    HEADERS

        Storages
        TestStorages
        StorageTemplateTests
        
    SOURCES

        AllStorageTest
        TypedStorageTest

        COOStorageTest
        CSRStorageTest
        ELLStorageTest
        DIAStorageTest
        JDSStorageTest
        SparseAssemblyStorageTest
        DenseStorageTest

        lamaStorageTest
    )

### checks ####
if ( ${SCAI_ASSERT_LEVEL} STREQUAL "OFF" )
   message ( STATUS "WARNING: tests will not check for exceptions (assertions are switched off)" )
endif ( ${SCAI_ASSERT_LEVEL} STREQUAL "OFF" )

### set include directories ###
include_directories ( ${SCAI_BOOST_INCLUDE_DIR} )

### add definitions ###
add_definitions ( -DLAMA_TESTFILE_PATH="${CMAKE_CURRENT_SOURCE_DIR}/../../../res/testfiles" )

### add link directories ###
if ( WIN32 )
    link_directories ( ${Boost_LIBRARY_DIRS} )
endif ( WIN32 )

### add executeable ###
add_executable ( lamaStorageTest ${CXX_SOURCES} )

### target link libraries ###
if ( WIN32 )
    target_link_libraries ( lamaStorageTest ${PROJECT_NAME} )
else ( WIN32 )
    target_link_libraries ( lamaStorageTest ${PROJECT_NAME} ${Boost_UNIT_TEST_FRAMEWORK_LIBRARY} ${Boost_REGEX_LIBRARY} )
endif ( WIN32 )
